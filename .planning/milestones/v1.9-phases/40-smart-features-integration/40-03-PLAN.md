Update the handleLogMeal function to handle three save paths:

```typescript
const handleLogMeal = useCallback(async () => {
  if (!canLog || isSubmitting) return;
  setIsSubmitting(true);
  setError(null);

  try {
    const mealFoodInputs: MealFoodInput[] = foods.map(f => ({
      foodId: f.foodId,
      foodName: f.foodName,
      grams: f.grams,
      proteinPer100g: f.proteinPer100g,
      carbsPer100g: f.carbsPer100g,
      fatPer100g: f.fatPer100g,
    }));

    // Compute summed macros for library save
    let totalProtein = 0, totalCarbs = 0, totalFat = 0;
    for (const f of mealFoodInputs) {
      totalProtein += (f.grams / 100) * f.proteinPer100g;
      totalCarbs += (f.grams / 100) * f.carbsPer100g;
      totalFat += (f.grams / 100) * f.fatPer100g;
    }

    if (mode === 'edit' && editMealId != null) {
      // Edit mode — update existing meal
      await foodsDb.updateMealWithFoods(editMealId, description, mealType!, mealFoodInputs, loggedAt);
    } else if (mode === 'library') {
      // Library mode — save to library only, no daily log (per D-15)
      await macrosDb.addLibraryMeal(description, mealType!, {
        protein: totalProtein,
        carbs: totalCarbs,
        fat: totalFat,
      });
    } else {
      // Normal mode — log to daily
      await foodsDb.addMealWithFoods(description, mealType!, mealFoodInputs, loggedAt);
      // Also save to library if toggle is on (per D-16)
      if (saveToLibrary) {
        await macrosDb.addLibraryMeal(description, mealType!, {
          protein: totalProtein,
          carbs: totalCarbs,
          fat: totalFat,
        });
      }
    }

    Vibration.vibrate(10);
    navigation.goBack();
  } catch (err) {
    setError(err instanceof Error ? err.message : "Couldn't save meal. Try again.");
    setIsSubmitting(false);
  }
}, [canLog, isSubmitting, foods, description, mealType, loggedAt, navigation, mode, editMealId, saveToLibrary]);
```
