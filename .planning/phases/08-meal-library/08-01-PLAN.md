   ```typescript
   function rowToLibraryMeal(row: {
     id: number;
     name: string;
     protein_grams: number;
     meal_type: string;
     created_at: string;
   }): LibraryMeal {
     return {
       id: row.id,
       name: row.name,
       proteinGrams: row.protein_grams,
       mealType: row.meal_type as MealType,
       createdAt: row.created_at,
     };
   }
   ```

   Then add the three CRUD functions:

   c. `deleteLibraryMeal(id: number): Promise<void>` -- DELETE FROM meal_library WHERE id = ?.
